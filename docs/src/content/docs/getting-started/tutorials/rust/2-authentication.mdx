In [Part 1](/getting-started/tutorials/rust/1-introduction) you scaffolded the
CLI and wired up the Nhost client. A command-line tool runs, exits, and runs
again — so to stay signed in between invocations we need to **persist the session
to disk**. The Rust SDK makes this a one-liner.

:::note
This is **Part 2** of the [Notes CLI with Nhost and Rust](/getting-started/tutorials/rust/1-introduction)

## Persist the session

The client accepts a `storage` backend that it uses to save and load the
session. The SDK ships `session::FileStorage`, which writes the session to a
JSON file. Update `make_client` to use it:

```rust title="src/main.rs"
use std::path::PathBuf;

use nhost::session::FileStorage;

fn make_client() -> NhostClient {
    let path = session_path();
    if let Some(parent) = path.parent() {
        let _ = std::fs::create_dir_all(parent);
    }
    create_client(Options {
        subdomain: Some(env("NHOST_SUBDOMAIN", "local")),
        region: Some(env("NHOST_REGION", "local")),
        storage: Some(Box::new(FileStorage::new(path))),
        ..Default::default()
    })
}

fn session_path() -> PathBuf {
    if let Ok(p) = std::env::var("NHOST_NOTES_SESSION") {
        return PathBuf::from(p);
    }
    let base = std::env::var("XDG_CONFIG_HOME")
        .map(PathBuf::from)
        .unwrap_or_else(|_| PathBuf::from(std::env::var("HOME").unwrap_or_default()).join(".config"));
    base.join("nhost-notes").join("session.json")
}
```

With a `storage` backend configured, the SDK's middleware does the rest: after
you sign in it saves the session, attaches the access token to every request,
and refreshes the token in the background when it expires. Your commands never
touch tokens directly.

## Add the auth commands

Add a small argument helper and the four authentication commands. The request
types come from `nhost::auth`:

```rust title="src/main.rs"
use nhost::auth::{SignInEmailPasswordRequest, SignOutRequest, SignUpEmailPasswordRequest};

fn need(args: &[String], n: usize, hint: &str) -> Result<()> {
    if args.len() < n {
        return Err(format!("expected {hint}").into());
    }
    Ok(())
}

async fn signup(client: &NhostClient, args: &[String]) -> Result<()> {
    need(args, 2, "signup <email> <password>")?;
    client
        .auth
        .sign_up_email_password(
            SignUpEmailPasswordRequest {
                email: args[0].clone(),
                password: args[1].clone(),
                options: None,
                code_challenge: None,
            },
            None,
        )
        .await?;
    if client.get_user_session().is_some() {
        println!("signed up and logged in as {}", args[0]);
    } else {
        println!("signed up; verify your email, then `login`");
    }
    Ok(())
}

async fn login(client: &NhostClient, args: &[String]) -> Result<()> {
    need(args, 2, "login <email> <password>")?;
    client
        .auth
        .sign_in_email_password(
            SignInEmailPasswordRequest { email: args[0].clone(), password: args[1].clone() },
            None,
        )
        .await?;
    println!("logged in as {}", args[0]);
    Ok(())
}

async fn logout(client: &NhostClient) -> Result<()> {
    if let Some(sess) = client.get_user_session() {
        let _ = client
            .auth
            .sign_out(
                SignOutRequest { refresh_token: Some(sess.session.refresh_token), all: None },
                None,
            )
            .await;
    }
    client.clear_session();
    println!("logged out");
    Ok(())
}

fn whoami(client: &NhostClient) -> Result<()> {
    match client.get_user_session().and_then(|s| s.session.user) {
        Some(u) => {
            println!("{} ({})", u.email.unwrap_or_default(), u.id);
            Ok(())
        }
        None => Err("not logged in".into()),
    }
}
```

`get_user_session` reads the current session from storage without a network
call — perfect for `whoami`. `clear_session` removes the cached file (a
client-side sign out), while `sign_out` revokes the refresh token on the server.

Wire the commands into `run`:

```rust title="src/main.rs"
    match cmd {
        "signup" => signup(&client, args).await,
        "login" => login(&client, args).await,
        "logout" => logout(&client).await,
        "whoami" => whoami(&client),
        "help" | "-h" | "--help" => {
            usage();
            Ok(())
        }
        other => {
            usage();
            Err(format!("unknown command {other:?}").into())
        }
    }
```

## Enable email/password sign-up

New projects require email verification by default. For a smooth CLI experience
during development, you can allow users to sign in immediately after sign-up:
in the Nhost dashboard open **Settings → Sign-In methods** and, under **Email
and Password**, disable *"Require verified emails"* (or verify via the email link
before running `login`).

```rust title="src/main.rs" collapse={9-21, 133-139} ins={1-4, 25-28, 41-44, 48, 53-61, 67-132}
use std::path::PathBuf;

use nhost::auth::{SignInEmailPasswordRequest, SignOutRequest, SignUpEmailPasswordRequest};
use nhost::session::FileStorage;
use nhost::{create_client, NhostClient, Options};

type Result<T> = std::result::Result<T, Box<dyn std::error::Error>>;

#[tokio::main]
async fn main() {
    let args: Vec<String> = std::env::args().skip(1).collect();
    if args.is_empty() {
        usage();
        std::process::exit(2);
    }
    if let Err(e) = run(&args[0], &args[1..]).await {
        eprintln!("error: {e}");
        std::process::exit(1);
    }
}

async fn run(cmd: &str, args: &[String]) -> Result<()> {
    let client = make_client();
    match cmd {
        "signup" => signup(&client, args).await,
        "login" => login(&client, args).await,
        "logout" => logout(&client).await,
        "whoami" => whoami(&client),
        "help" | "-h" | "--help" => {
            usage();
            Ok(())
        }
        other => {
            usage();
            Err(format!("unknown command {other:?}").into())
        }
    }
}

fn make_client() -> NhostClient {
    let path = session_path();
    if let Some(parent) = path.parent() {
        let _ = std::fs::create_dir_all(parent);
    }
    create_client(Options {
        subdomain: Some(env("NHOST_SUBDOMAIN", "local")),
        region: Some(env("NHOST_REGION", "local")),
        storage: Some(Box::new(FileStorage::new(path))),

}

}

fn env(key: &str, fallback: &str) -> String {
    std::env::var(key).unwrap_or_else(|_| fallback.to_string())
}
```

## Try it

```bash Terminal
cargo run -- signup ada@example.com 'not-a-secret'
cargo run -- whoami
# ada@example.com (a1b2c3d4-...)

cargo run -- logout
cargo run -- whoami
# error: not logged in

cargo run -- login ada@example.com 'not-a-secret'
```

The session is written to `~/.config/nhost-notes/session.json`. Delete that file
(or run `logout`) to sign out.
