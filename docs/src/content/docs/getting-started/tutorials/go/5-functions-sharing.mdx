## Share a note with a collaborator

Add a `note_collaborators` table in the **SQL Editor**:

```sql SQL Editor
CREATE TABLE public.note_collaborators (
  note_id uuid NOT NULL REFERENCES public.notes (id) ON DELETE CASCADE,
  user_id uuid NOT NULL REFERENCES auth.users (id) ON DELETE CASCADE,
  role    text NOT NULL DEFAULT 'viewer' CHECK (role IN ('viewer', 'editor')),
  PRIMARY KEY (note_id, user_id)
);
```

Track it and add a `collaborators` relationship on `notes`. Now update the
`notes` permissions so collaborators can see and edit shared notes — this is the
one place the owner-only rule opens up:

- **notes → select** filter — owner **or** any collaborator:

  ```json
  { "_or": [
      { "user_id": { "_eq": "X-Hasura-User-Id" } },
      { "collaborators": { "user_id": { "_eq": "X-Hasura-User-Id" } } }
  ]}
  ```
- **notes → update** filter — owner **or** an `editor` collaborator:

  ```json
  { "_or": [
      { "user_id": { "_eq": "X-Hasura-User-Id" } },
      { "collaborators": { "_and": [
          { "user_id": { "_eq": "X-Hasura-User-Id" } },
          { "role": { "_eq": "editor" } }
      ]}}
  ]}
  ```

For **note_collaborators**, let a note's owner manage shares: use the `insert`
check / `select` / `delete` filter
`{ "note": { "user_id": { "_eq": "X-Hasura-User-Id" } } }`.

Add the commands (the owner shares by the collaborator's user id):

```go title="main.go"
func cmdShare(ctx context.Context, c *nhost.Client, args []string) error {
	fs := flag.NewFlagSet("share", flag.ContinueOnError)
	role := fs.String("role", "viewer", "viewer|editor")
	if err := fs.Parse(args); err != nil {
		return err
	}
	if fs.NArg() < 2 {
		return errors.New("share [--role viewer|editor] <noteId> <userId>")
	}
	_, err := gql(ctx, c, `
		mutation Share($noteId: uuid!, $userId: uuid!, $role: String!) {
			insert_note_collaborators_one(
				object: {note_id: $noteId, user_id: $userId, role: $role}
				on_conflict: {constraint: note_collaborators_pkey, update_columns: [role]}
			) { note_id role }
		}`, graphql.Variables{"noteId": fs.Arg(0), "userId": fs.Arg(1), "role": *role})
	if err != nil {
		return err
	}
	fmt.Printf("shared %s with %s as %s\n", fs.Arg(0), fs.Arg(1), *role)
	return nil
}
```

Wire it with `case "share": return cmdShare(ctx, c, args)` (an `unshare` command
deletes the row — see the complete example).

## The export function

Serverless functions run on the backend with your project. Create
`functions/notes/export.ts` in your Nhost project (functions are deployed from
the project's `functions/` directory — see the
[Functions docs](/products/functions)). It reads the caller's notes over
GraphQL, renders Markdown, and uploads the result to Storage — **forwarding the
caller's token**, so Hasura permissions and Storage rules apply exactly as they
do in the CLI:

```typescript title="functions/notes/export.ts"
import type { Request, Response } from "express";
import { createClient } from "@nhost/nhost-js";

const nhost = createClient({
  region: process.env.NHOST_REGION,
  subdomain: process.env.NHOST_SUBDOMAIN,
});

export default async (req: Request, res: Response) => {
  const authorization = req.headers.authorization ?? "";
  if (!authorization) {
    return res.status(401).json({ error: "Missing Authorization header" });
  }

  const { body } = await nhost.graphql.request<{ notes: { title: string; content: string }[] }>(
    { query: `query { notes(order_by: {updated_at: desc}) { title content } }` },
    { headers: { Authorization: authorization } },
  );

  const notes = body.data?.notes ?? [];
  const document =
    `# Notes export\n\n_${notes.length} note(s)_\n\n` +
    notes.map((n) => `# ${n.title || "Untitled"}\n\n${n.content}`).join("\n\n---\n\n");

  const fileName = `export-${Date.now()}.md`;
  const upload = await nhost.storage.uploadFiles(
    { "bucket-id": "notes", "file[]": [new File([document], fileName, { type: "text/markdown" })] },
    { headers: { Authorization: authorization } },
  );

  const file = upload.body.processedFiles?.[0];
  res.status(200).json({ fileId: file?.id, fileName, noteCount: notes.length });
};
```

## Call it from the CLI

`Functions.Post` sends an authenticated `POST` — the SDK attaches your token, so
the function receives it in the `Authorization` header:

```go title="main.go"
func cmdExport(ctx context.Context, c *nhost.Client) error {
	resp, err := c.Functions.Post(ctx, "/notes/export", struct{}{}, nil)
	if err != nil {
		return err
	}
	out, _ := json.MarshalIndent(resp.Body, "", "  ")
	fmt.Println(string(out))
	return nil
}
```

Add `case "export": return cmdExport(ctx, c)` and the `encoding/json` import.

## Try it

```bash Terminal
# share a note with another user (get their id from `whoami` while signed in as them)
go run . share --role editor <noteId> <otherUserId>

# export your notes to a Markdown file in Storage
go run . export
# { "fileId": "...", "fileName": "export-1737045.md", "noteCount": 3 }

go run . download <fileId> ./notes.md
```

## What you built

You now have a working notes CLI that exercises every core Nhost product:

- **Auth** with a persisted session that survives between runs
- **GraphQL** CRUD protected by owner-scoped row permissions
- **Storage** for attachments, access-controlled by the same rules
- **Functions** that reuse the caller's identity to combine products server-side

The full, runnable source — including `note show`, `note edit/mv/rm`, `notebook`,
`tag ls`, `unshare`, and richer output — is the
[`notes-cli` example](https://github.com/nhost/nhost/tree/main/packages/nhost-go/examples/notes-cli)
on GitHub.
