## Configuration

```js
// {project-root}/helpers/with-nhost.js
import { configureNhostSSR } from '@nhost/nextjs'

export const withNhost = configureNhostSSR({
  backendUrl: 'http://127.0.0.1:1337'
})
```

## Usage

```js
// {project-root}/pages/my-page.jsx
import { withNhost } from '../helpers/with-nhost'

import { useAccessToken, useAuthenticated } from '@nhost/react'

const MyPage = () => {
  const isAuthenticated = useAuthenticated()
  const jwt = useAccessToken()
  if (isAuthenticated) {
    return <div>Non authenticated query</div>
  } else {
    return <div>Authenticated user. Here's the JWT to use: {jwt}</div>
  }
}

export default withNhost(MyPage)
```
